44  	/**

46  	 *

48  	 * 			The log level

51  	 * @param 	throwable

53  	 */
54  	@Override
55  	protected void logImpl(LogLevel level,Object[] message,Throwable throwable)
56  	{
57  		//format log message (without exception)
58  		String text=this.formatLogMessage(level,message,null);
59  
60  		//get log level
61  		int levelValue=level.getValue();
62  		switch(levelValue)
63  		{
64  			case LogLevel.DEBUG_LOG_LEVEL_VALUE:
65  				this.APACHE_LOGGER.debug(text);
66  				break;
67  			case LogLevel.ERROR_LOG_LEVEL_VALUE:
68  				this.APACHE_LOGGER.error(text,throwable);
69  				break;
70  			case LogLevel.INFO_LOG_LEVEL_VALUE:
71  			default:
72  				this.APACHE_LOGGER.info(text);
