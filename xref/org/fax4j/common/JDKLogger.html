11  /**
12   * This is a simple proxy implementation of the logger that transfers
13   * all log events to the JDK logger.
14   * 
15   * @author 	Sagie Gur-Ari
16   * @version 1.03
17   * @since	0.27
18   */
19  public class JDKLogger extends AbstractLogger
20  {
21  	/**The JDK logger*/
22  	protected final Logger JDK_LOGGER;
23  	
24  	/**
25  	 * This is the default constructor.
26  	 */
27  	public JDKLogger()
28  	{
29  		super();
30  
31  		//init logger
32  		this.JDK_LOGGER=JDKLogger.initializeLogger();
33  	}
34  	
35  	/**
36  	 * This function initializes and returns the logger.
37  	 * 
38  	 * @return	The logger
39  	 */
40  	protected static final Logger initializeLogger()
41  	{
42  		//get logger
43  		Logger logger=Logger.getLogger(FaxClient.class.getName());
44  		
45  		//enable all log events (fax4j logger filters out uneeded log events)
46  		logger.setLevel(Level.ALL);
47  		logger.setFilter(null);
48  		
49  		//enable to pass log events to parent loggers
50  		logger.setUseParentHandlers(true);
51  
52  		//create handler
53  		Formatter formatter=new SimpleFormatter();
54  		Handler handler=new StreamHandler(System.out,formatter);
55  		
56  		//set filtering
57  		handler.setLevel(logger.getLevel());
58  		handler.setFilter(logger.getFilter());
59  
60  		//add handler
61  		logger.addHandler(handler);
62  
63  		return logger;
64  	}
65  
66  	/**
67  	 * Logs the provided data.
68  	 *  
69  	 * @param 	level
70  	 * 			The log level
71  	 * @param	message
72  	 * 			The message parts (may be null)
73  	 * @param 	throwable
74  	 * 			The error (may be null)
75  	 */
76  	@Override
77  	protected void logImpl(LogLevel level,Object[] message,Throwable throwable)
78  	{
79  		//get log level
80  		int levelValue=level.getValue();
81  		Level jdkLevel=null;
82  		switch(levelValue)
83  		{
84  			case LogLevel.DEBUG_LOG_LEVEL_VALUE:
85  				jdkLevel=Level.FINEST;
86  				break;
87  			case LogLevel.ERROR_LOG_LEVEL_VALUE:
88  				jdkLevel=Level.SEVERE;
89  				break;
90  			case LogLevel.INFO_LOG_LEVEL_VALUE:
91  			default:
92  				jdkLevel=Level.FINE;
93  				break;
94  		}
95  
96  		if(jdkLevel!=null)
97  		{
98  			//format log message (without exception)
99  			String text=this.formatLogMessage(level,message,null);
100 
101 			//log
102 			this.JDK_LOGGER.log(jdkLevel,text,throwable);
103 		}
104 	}
