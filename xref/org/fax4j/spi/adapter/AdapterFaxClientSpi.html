13  /**
14   * This class implements the fax client service provider interface.<br>
15   * This implementation will proxy the requests to another internal SPI.<br>
16   * The internal SPI created will be based on configuration in the fax4j properties.<br>
17   * The configuration contains conditions such as OS, availability of java classes,
18   * availability of native libraries, availability of executables or validation that
19   * a certain fax4j property is defined and contains a value.<br>
20   * Once all conditions are met the internal SPI is created, otherwise the next SPI configuration
21   * is used.<br>
22   * If no SPI conditions are met, an exception will be thrown.<br>
23   * <br>
24   * The configuration of the fax4j framework is made up of 3 layers.<br>
25   * The configuration is based on simple properties.<br>
26   * Each layer overrides the lower layers by adding/changing the property values.<br>
27   * The first layer is the internal fax4j.properties file located in the fax4j jar.<br>
28   * This layer contains the preconfigured values for the fax4j framework and can be changed
29   * by updating these properties in the higher layers.<br>
30   * The second layer is the external fax4j.properties file that is located on the classpath.<br>
31   * This file is optional and provides the ability to override the internal configuration for the
32   * entire fax4j framework.<br>
33   * The top most layer is the optional java.util.Properties object provided by the external classes
34   * when creating a new fax client.<br>
35   * These properties enable to override the configuration of the lower 2 layers.<br>
36   * <br>

344 	/**
345 	 * Validates the SPI condition.
346 	 *  
347 	 * @param 	key
348 	 * 			The condition key <br>
349 	 * 			property - checks a fax4j property is defined and contains a value<br>
350 	 * 			OS - checks OS name contains the value (case insensitive)<br>
351 	 * 			java-class - checks that java class can be loaded<br>
352 	 * 			native-lib - checks that native lib can be loaded<br>
353 	 * 			executable - checks executable is on system path<br>
354 	 * 			stable - checks for a fax4j property org.fax4j.spi.xxx.stable is defined and equals true, where xxx is the SPI key (for example: org.fax4j.spi.adapter.stable=true)
355 	 * @param 	value
356 	 * 			The condition value
357 	 * @return	True if the condition is met
358 	 */
359 	protected boolean validateCondition(String key,String value)
360 	{
361 		boolean valid=false;
362 		FaxClientSpiConfigurationConstants condition=FaxClientSpiConfigurationConstants.getEnum(key);
363 		String propertyValue=null;
364 		switch(condition)
365 		{
366 			case PROPERTY_CONDITION:
367 				//get value
368 				propertyValue=this.getConfigurationValue(value);
369 				
370 				if(propertyValue!=null)
371 				{
372 					valid=true;
373 				}
374 				break;
375 			case OS_CONDITION:
376 				//get OS name
377 				String osName=System.getProperty("os.name");
378 				
379 				//change values to lower case
380 				osName=osName.toLowerCase();
381 				String updatedValue=value.toLowerCase();
382 				
383 				if(osName.contains(updatedValue))
384 				{
385 					valid=true;
386 				}
387 				break;
388 			case JAVA_CLASS_CONDITION:
389 				try
390 				{
391 					//load class
392 					ReflectionHelper.getType(value);
393 					
394 					valid=true;
395 				}
396 				catch(Throwable throwable)
397 				{
398 					//ignore
399 				}
400 				break;
401 			case NATIVE_LIB_CONDITION:
402 				try
403 				{
404 					//load library
405 					System.loadLibrary(value);
406 					
407 					valid=true;
408 				}
409 				catch(Throwable throwable)
410 				{
411 					//ignore
412 				}
413 				break;
414 			case EXECUTABLE_CONDITION:
415 				//get system path
416 				String systemPath=System.getProperty("java.library.path");
417 				String[] systemPathElements=systemPath.split(System.getProperty("path.separator"));
418 				
419 				//get amount
420 				int amount=systemPathElements.length;
421 				
422 				String directoryPath=null;
423 				File file=null;
424 				for(int index=0;index<amount;index++)
425 				{
426 					//get next element
427 					directoryPath=systemPathElements[index];
428 
429 					//get file
430 					file=new File(directoryPath,value);
431 					
432 					if((file.exists())&&(file.isFile()))
433 					{
434 						valid=true;
435 					}
436 				}
437 				break;
438 			case STABLE_CONDITION:
439 				//get property key
440 				String propertyKey=FaxClientSpiConfigurationConstants.SPI_STABLE_PROPERTY_KEY_PREFIX+value+FaxClientSpiConfigurationConstants.SPI_STABLE_PROPERTY_KEY_SUFFIX;
441 				
442 				//get value
443 				propertyValue=this.getConfigurationValue(propertyKey);
444 				
445 				valid=Boolean.parseBoolean(propertyValue);
446 				break;
447 			default:
448 				throw new FaxException("Invalid condition key provided: "+key);
449 		}
450 		
451 		return valid;
452 	}
