59  	protected Transport createTransport(Session session)
60  	{
61  		Transport transport=null;
62  		try
63  		{
64  			//get the transport
65  			transport=session.getTransport(this.transportProtocol);
66  
67  			//connect
68  			if(this.transportPort>0)
69  			{
70  				transport.connect(this.transportHost,this.transportPort,this.userName,this.password);
71  			}
72  			else
73  			{
74  				transport.connect(this.transportHost,this.userName,this.password);
75  			}
76  		}
77  		catch(Exception exception)
78  		{
79  			throw new FaxException("Error while connecting to mail host: "+this.transportHost,exception);
80  		}
81  		
82  		return transport;
83  	}
