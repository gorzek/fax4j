38  	 * This function creates the HTTP client configuration to be

40  	 *

43  	 * @return	The HTTP client configuration
44  	 */
45  	public HTTPClientConfiguration createHTTPClientConfiguration(ConfigurationHolder configurationHolder)
46  	{
47  		//get server values
48  		String hostName=configurationHolder.getConfigurationValue(HTTPClientConfigurationConstants.HOST_NAME_PROPERTY_KEY);
49  		if(hostName==null)
50  		{
51  			throw new FaxException("Host name not defined in property: "+HTTPClientConfigurationConstants.HOST_NAME_PROPERTY_KEY);
52  		}
53  		String value=configurationHolder.getConfigurationValue(HTTPClientConfigurationConstants.PORT_PROPERTY_KEY);
54  		if(value==null)
55  		{
56  			value=String.valueOf(-1);
57  		}
58  		int port=Integer.parseInt(value);
59  		boolean ssl=Boolean.parseBoolean(configurationHolder.getConfigurationValue(HTTPClientConfigurationConstants.SSL_PROPERTY_KEY));
60  		
61  		//create configuration
62  		CommonHTTPClientConfiguration configuration=new CommonHTTPClientConfiguration();
63  		
64  		//set values
65  		configuration.setHostName(hostName);
66  		configuration.setPort(port);
67  		configuration.setSSL(ssl);
68  
69  		//set methods
70  		Enum<?>[] methodProperties=new Enum<?>[]{HTTPClientConfigurationConstants.SUBMIT_HTTP_METHOD_PROPERTY_KEY,
71  												 HTTPClientConfigurationConstants.SUSPEND_HTTP_METHOD_PROPERTY_KEY,
72  												 HTTPClientConfigurationConstants.RESUME_HTTP_METHOD_PROPERTY_KEY,
73  												 HTTPClientConfigurationConstants.CANCEL_HTTP_METHOD_PROPERTY_KEY,
74  												 HTTPClientConfigurationConstants.GET_STATUS_HTTP_METHOD_PROPERTY_KEY};
75  		FaxActionType[] faxActionTypes=new FaxActionType[]{FaxActionType.SUBMIT_FAX_JOB,
76  														   FaxActionType.SUSPEND_FAX_JOB,
77  														   FaxActionType.RESUME_FAX_JOB,
78  														   FaxActionType.CANCEL_FAX_JOB,
79  														   FaxActionType.GET_FAX_JOB_STATUS};
80  		HTTPMethod httpMethod=null;
81  		for(int index=0;index<methodProperties.length;index++)
82  		{
83  			//set next method
84  			value=configurationHolder.getConfigurationValue(methodProperties[index]);
85  			httpMethod=HTTPMethod.POST;
86  			if(value!=null)
87  			{
88  				httpMethod=HTTPMethod.valueOf(value);
89  			}
90  			configuration.setMethod(faxActionTypes[index],httpMethod);
91  		}
92  
93  		return configuration;
94  	}
